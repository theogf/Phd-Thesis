\begin{algorithmic}[1]

\Procedure{SUM}{ $\{x\}$}

\State $y\gets0$
\For{$i \gets 1 : N^{x}$} \Comment{Time series $\{x\}$ has length $N^{x}$}
   \State $y\gets y+x(i)$ \Comment{Summing up.}
\EndFor

\State \textbf{return}  $y$ 
\EndProcedure
\end{algorithmic}
